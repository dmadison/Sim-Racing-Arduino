28// This example requires the Arduino Joystick Library
29// Download Here: https://github.com/MHeironimus/ArduinoJoystickLibrary

31#include <SimRacing.h>
32#include <Joystick.h>
33
34// Set this option to 'true' to send the shifter's X/Y position
35// as a joystick. This is not needed for most games.
36const bool SendAnalogAxis = false;
37
38// Set this option to 'true' to send the raw state of the reverse
39// trigger as its own button. This is not needed for any racing
40// games, but can be useful for custom controller purposes.
41const bool SendReverseRaw = false;
42
43const int Pin_ShifterX = A0;
44const int Pin_ShifterY = A2;
45const int Pin_ShifterRev = 2;
46
47SimRacing::LogitechShifter shifter(Pin_ShifterX, Pin_ShifterY, Pin_ShifterRev);
48
49const int Gears[] = { 1, 2, 3, 4, 5, 6, -1 };
50const int NumGears = sizeof(Gears) / sizeof(Gears[0]);
51
52const int ADC_Max = 1023; // 10-bit on AVR
53
54Joystick_ Joystick(
55 JOYSTICK_DEFAULT_REPORT_ID, // default report (no additional pages)
56 JOYSTICK_TYPE_JOYSTICK, // so that this shows up in Windows joystick manager
57 NumGears + SendReverseRaw, // number of buttons (7 gears: reverse and 1-6)
58 0, // number of hat switches (none)
59 SendAnalogAxis, SendAnalogAxis, // include X and Y axes for analog output, if set above
60 false, false, false, false, false, false, false, false, false); // no other axes
61
62void updateJoystick(); // forward-declared function for non-Arduino environments
63
64
65void setup() {
66 shifter.begin();
67
68 // if you have one, your calibration line should go here
69
70 Joystick.begin(false); // 'false' to disable auto-send
71 Joystick.setXAxisRange(0, ADC_Max);
72 Joystick.setYAxisRange(ADC_Max, 0); // invert axis so 'up' is up
73
74 updateJoystick(); // send initial state
75}
76
77void loop() {
78 shifter.update();
79
80 if (SendAnalogAxis == true || shifter.gearChanged()) {
81 updateJoystick();
82 }
83}
84
85void updateJoystick() {
86 // set the buttons corresponding to the gears
87 for (int i = 0; i < NumGears; i++) {
88 if (shifter.getGear() == Gears[i]) {
89 Joystick.pressButton(i);
90 }
91 else {
92 Joystick.releaseButton(i);
93 }
94 }
95
96 // set the analog axes (if the option is set)
97 if (SendAnalogAxis == true) {
98 int x = shifter.getPosition(SimRacing::X, 0, ADC_Max);
99 int y = shifter.getPosition(SimRacing::Y, 0, ADC_Max);
100 Joystick.setXAxis(x);
101 Joystick.setYAxis(y);
102 }
103
104 // set the reverse button (if the option is set)
105 if (SendReverseRaw == true) {
106 bool reverseState = shifter.getReverseButton();
107 Joystick.setButton(NumGears, reverseState); // "NumGears" is the 0-indexed max gear + 1
108 }
109
110 Joystick.sendState();
111}
