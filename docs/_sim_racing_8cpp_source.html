65static long remap(long value, long inMin, long inMax, long outMin, long outMax) {
66 // if inverted, swap min/max and adjust position of value
67 if (inMin > inMax) {
68 const long temp = inMin;
69 inMin = inMax;
70 inMax = temp;
71
72 value = invertAxis(value, inMin, inMax);
73 }
74
75 if (value <= inMin) return outMin;
76 if (value >= inMax) return outMax;
77 return map(value, inMin, inMax, outMin, outMax);
78}

110static void waitClient(Stream& client) {
111 flushClient(client);

113}

127static void readFloat(float& value, Stream& client) {
128 client.print("(to skip this step and go with the default value of '");
129 client.print(value);
130 client.print("', send 'n')");
131 client.println();
132
133 waitClient(client);
134 if (client.peek() == 'n') return; // skip this step
135
136 float input;
137
138 while (true) {
139 client.setTimeout(200);
140 input = client.parseFloat();
141
142 if (input >= 0.0 && input <= 1.0) {
143 client.print(F("Set the new value to '"));
144 client.print(input);
145 client.println("'");
146 break;
147 }
148 client.print(F("Input '"));
149 client.print(input);
150 client.print(F("' not within acceptable range (0.0 - 1.0). Please try again."));
151 client.println();
152
153 waitClient(client);
154 }
155
156 value = input;
157}

164DeviceConnection::DeviceConnection(uint8_t pin, bool invert, unsigned long detectTime)
165 :
166 Pin(pin), Inverted(invert), stablePeriod(detectTime), // constants(ish)
167
168 /* Assume we're connected on first call
169 */
170 state(ConnectionState::Connected),
171
172 /* Init state to "not inverted", which is the connected state. For example
173 * if we're looking for 'HIGH' then inverted is false, which means the
174 * initial state is 'true' and thus connected.
175 *
176 * We're assuming we're connected on first call here because it allows
177 * the device to be read as connected as soon as the board turns on, without
178 * having to wait an arbitrary amount.
179 */
180 pinState(!Inverted),
181
182 /* Set the last pin change to right now minus the stable period so it's
183 * read as being already stable. Again, this will make the class return
184 * 'present' as soon as the board starts up
185 */
186 lastChange(millis() - detectTime)
187
188{
189 pinMode(Pin, INPUT); // set pin as input, *no* pull-up
190}
191
192void DeviceConnection::poll() {
193 const bool newState = readPin();
194
195 if (newState == HIGH && state == ConnectionState::Connected) return; // short circuit, already connected
196
197 // check if the pin changed. if it did, record the time
198 if (pinState != newState) {
199 pinState = newState;
200 lastChange = millis();
201
202 // rising, we just connected
203 if (pinState == HIGH) {
204 state = ConnectionState::PlugIn;
205 }
206 // falling, we just disconnected
207 else {
208 state = ConnectionState::Unplug;
209 }
210 }
211
212 // if pin hasn't changed, compare stable times
213 else {
214 // check stable connection (over time)
215 if (pinState == HIGH) {
216 const unsigned long now = millis();
217 if (now - lastChange >= stablePeriod) {
218 state = ConnectionState::Connected;
219 }
220 }
221 // if we were previously unplugged and are still low, now we're disconnected
222 else if (state == ConnectionState::Unplug) {
223 state = ConnectionState::Disconnected;
224 }
225 }
226}
227
228DeviceConnection::ConnectionState DeviceConnection::getState() const {
229 return state;
230}
231
232bool DeviceConnection::isConnected() const {
233 return this->getState() == ConnectionState::Connected;
234}
235
236void DeviceConnection::setStablePeriod(unsigned long t) {
237 stablePeriod = t;
238
239 if (state == ConnectionState::Connected) {
240 const unsigned long now = millis();
241
242 // if we were previously considered connected, adjust the timestamps
243 // accordingly so that we still are
244 if (now - lastChange < stablePeriod) {
245 lastChange = now - stablePeriod;
246 }
247 }
248}

321bool Pedals::update() {
322 bool changed = false;
323
324 detector.poll();
325 if (detector.getState() == DeviceConnection::Connected) {
326 // if connected, read all pedal positions
327 for (int i = 0; i < getNumPedals(); ++i) {
328 changed |= pedalData[i].read();
329 }
330 }
331 else if (detector.getState() == DeviceConnection::Unplug) {
332 // on unplug event, zero all pedals
333 for (int i = 0; i < getNumPedals(); ++i) {
334 const int min = pedalData[i].getMin();
335 pedalData[i].setPosition(min);
336 }
337 }
338
339 return changed;
340}

598bool AnalogShifter::update() {
599 detector.poll();
600
601 switch (detector.getState()) {
602
603 // connected! poll the ADC for new analog axis data
604 case(DeviceConnection::Connected):
605 analogAxis[Axis::X].read();
606 analogAxis[Axis::Y].read();
607 break;
608
609 // on an unplug event, we want to reset our position back to
610 // neutral and then immediately return
611 case(DeviceConnection::Unplug):
612 {
613 const int8_t previousGear = this->getGear();
614
615 analogAxis[Axis::X].setPosition(calibration.neutralX);
616 analogAxis[Axis::Y].setPosition(calibration.neutralY);
617
618 if (previousGear != 0) changed = true;
619 currentGear = 0;
620 return changed;
621 break;
622 }
623
624 // if the device is either disconnected or just plugged in and unstable, set gear
625 // 'changed' to false and then immediately return false to save on processing
626 case(DeviceConnection::PlugIn):
627 case(DeviceConnection::Disconnected):
628 changed = false;
629 return changed;
630 break;
631 }
632
633 const int8_t previousGear = this->getGear();
634 const bool prevOdd = ((previousGear != -1) && (previousGear & 1)); // were we previously in an odd gear
635 const bool prevEven = (!prevOdd && previousGear != 0); // were we previously in an even gear
636
637 const int x = analogAxis[Axis::X].getPosition();
638 const int y = analogAxis[Axis::Y].getPosition();
639 int8_t newGear = 0;
640
641 // If we're below the 'release' thresholds, we must still be in the previous gear
642 if ((prevOdd && y > calibration.oddRelease) || (prevEven && y < calibration.evenRelease)) {
643 newGear = previousGear;
644 }
645
646 // If we're *not* below the release thresholds, we may be in a different gear
647 else {
648 // Check if we're in even or odd gears (Y axis)
649 if (y > calibration.oddTrigger) {
650 newGear = 1; // we're in an odd gear
651 }
652 else if (y < calibration.evenTrigger) {
653 newGear = 2; // we're in an even gear
654 }
655
656 if (newGear != 0) {
657 // Now check *which* gear we're in, if we're in one (X axis)
658 if (x > calibration.rightEdge) newGear += 4; // 1-2 + 4 = 5-6
659 else if (x >= calibration.leftEdge) newGear += 2; // 1-2 + 2 = 3-4
660 // (note the '>=', because it would normally be a '<' check for the lower range)
661 // else gear = 1-2 (as set above)
662
663 const bool reverse = getReverseButton();
664
665 // If the reverse button is pressed and we're in 5th gear
666 // something is wrong. Revert that and go back to neutral.
667 if (reverse && newGear == 5) {
668 newGear = 0;
669 }
670
671 // If the reverse button is pressed or we were previously
672 // in reverse *and* we are currently in 6th gear, then we
673 // should be in reverse.
674 else if ((reverse || previousGear == -1) && newGear == 6) {
675 newGear = -1;
676 }
677 }
678 }
679
680 changed = (newGear != previousGear) ? 1 : 0;
681 currentGear = newGear;
682
683 return changed;
684}

705void AnalogShifter::setCalibration(
706 GearPosition neutral,
707 GearPosition g1, GearPosition g2, GearPosition g3, GearPosition g4, GearPosition g5, GearPosition g6,
708 float engagePoint, float releasePoint, float edgeOffset) {
709
710 // limit percentage thresholds
711 engagePoint = floatPercent(engagePoint);
712 releasePoint = floatPercent(releasePoint);
713 edgeOffset = floatPercent(edgeOffset);
714
715 const int xLeft = (g1.x + g2.x) / 2; // find the minimum X position average
716 const int xRight = (g5.x + g6.x) / 2; // find the maximum X position average
717
718 const int yOdd = (g1.y + g3.y + g5.y) / 3; // find the maximum Y position average
719 const int yEven = (g2.y + g4.y + g6.y) / 3; // find the minimum Y position average
720
721 // set X/Y calibration and inversion
722 analogAxis[Axis::X].setCalibration({ xLeft, xRight });
723 analogAxis[Axis::Y].setCalibration({ yEven, yOdd });
724
725 // save neutral values (raw)
726 calibration.neutralX = neutral.x;
727 calibration.neutralY = neutral.y;
728
729 // get normalized and inverted neutral values
730 // this lets us take advantage of the AnalogInput normalization function
731 // that handles inverted axes and automatic range rescaling, so the rest of
732 // the calibration options can be in the normalized range
733 const Axis axes[2] = { Axis::X, Axis::Y };
734 int* const neutralAxis[2] = { &neutral.x, &neutral.y };
735 for (int i = 0; i < 2; i++) {
736 const int previous = analogAxis[axes[i]].getPositionRaw(); // save current value
737 analogAxis[axes[i]].setPosition(*neutralAxis[i]); // set new value to neutral calibration
738 *neutralAxis[i] = analogAxis[axes[i]].getPosition(); // get normalized neutral value
739 analogAxis[axes[i]].setPosition(previous); // reset axis position to previous
740 }
741
742 // calculate the distances between each neutral and the limits of each axis
743 const int yOddDiff = AnalogInput::Max - neutral.y;
744 const int yEvenDiff = neutral.y - AnalogInput::Min;
745
746 const int leftDiff = neutral.x - AnalogInput::Min;
747 const int rightDiff = AnalogInput::Max - neutral.x;
748
749 // calculate and save the trigger and release points for each level
750 calibration.oddTrigger = neutral.y + ((float)yOddDiff * engagePoint);
751 calibration.oddRelease = neutral.y + ((float)yOddDiff * releasePoint);
752
753 calibration.evenTrigger = neutral.y - ((float)yEvenDiff * engagePoint);
754 calibration.evenRelease = neutral.y - ((float)yEvenDiff * releasePoint);
755
756 calibration.leftEdge = neutral.x - ((float)leftDiff * edgeOffset);
757 calibration.rightEdge = neutral.x + ((float)rightDiff * edgeOffset);
758
759#if 0
760 Serial.print("Odd Trigger: ");
761 Serial.println(calibration.oddTrigger);
762 Serial.print("Odd Release: ");
763 Serial.println(calibration.oddRelease);
764 Serial.print("Even Trigger: ");
765 Serial.println(calibration.evenTrigger);
766 Serial.print("Even Release: ");
767 Serial.println(calibration.evenRelease);
768 Serial.print("Left Edge: ");
769 Serial.println(calibration.leftEdge);
770 Serial.print("Right Edge: ");
771 Serial.println(calibration.rightEdge);
772 Serial.println();
773
774 Serial.print("X Min: ");
775 Serial.println(analogAxis[Axis::X].getMin());
776 Serial.print("X Max: ");
777 Serial.println(analogAxis[Axis::X].getMax());
778
779 Serial.print("Y Min: ");
780 Serial.println(analogAxis[Axis::Y].getMin());
781 Serial.print("Y Max: ");
782 Serial.println(analogAxis[Axis::Y].getMax());
783#endif
784}

786void AnalogShifter::serialCalibration(Stream& iface) {
787 if (isConnected() == false) {
788 iface.print(F("Error! Cannot perform calibration, "));
789 iface.print(F("shifter"));
790 iface.println(F(" is not connected."));
791 return;
792 }
793
794 const char* separator = "------------------------------------";
795
796 iface.println();
797 iface.println(F("Sim Racing Library Shifter Calibration"));
798 iface.println(separator);
799 iface.println();
800
801 AnalogShifter::GearPosition gears[7]; // neutral, then 1-6
802 float engagementPoint = CalEngagementPoint;
803 float releasePoint = CalReleasePoint;
804 float edgeOffset = CalEdgeOffset;
805
806 for (int i = 0; i <= 6; i++) {
807 const String gearName = this->getGearString(i);
808
809 iface.print(F("Please move the gear shifter into "));
