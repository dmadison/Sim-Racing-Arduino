28// This example requires the Arduino Joystick Library
29// Download Here: https://github.com/MHeironimus/ArduinoJoystickLibrary
30
31#include <SimRacing.h>
32#include <Joystick.h>
33
34const int Pin_Gas = A2;
35const int Pin_Brake = A1;
36const int Pin_Clutch = A0;
37
38SimRacing::LogitechPedals pedals(Pin_Gas, Pin_Brake, Pin_Clutch);
39
40Joystick_ Joystick(
41 JOYSTICK_DEFAULT_REPORT_ID, // default report (no additional pages)
42 JOYSTICK_TYPE_JOYSTICK, // so that this shows up in Windows joystick manager
43 0, // number of buttons (none)
44 0, // number of hat switches (none)
45 false, false, // no X and Y axes
46 pedals.hasPedal(SimRacing::Clutch), // include Z axis for the clutch pedal
47 pedals.hasPedal(SimRacing::Brake), // include Rx axis for the brake pedal
48 pedals.hasPedal(SimRacing::Gas), // include Ry axis for the gas pedal
49 false, false, false, false, false, false); // no other axes
50
51const int ADC_Max = 1023; // max value of the analog inputs, 10-bit on AVR boards
52
53
54void setup() {
55 pedals.begin(); // initialize pedal pins
56
57 // if you have one, your calibration line should go here
58
59 Joystick.begin(false); // 'false' to disable auto-send
60
61 Joystick.setZAxisRange(0, ADC_Max);
62 Joystick.setRxAxisRange(0, ADC_Max);
63 Joystick.setRyAxisRange(0, ADC_Max);
64}
65
66void loop() {
67 pedals.update();
68
69 if (pedals.hasPedal(SimRacing::Gas)) {
70 int gasPedal = pedals.getPosition(SimRacing::Gas, 0, ADC_Max);
71 Joystick.setRyAxis(gasPedal);
72 }
73
74 if (pedals.hasPedal(SimRacing::Brake)) {
75 int brakePedal = pedals.getPosition(SimRacing::Brake, 0, ADC_Max);
76 Joystick.setRxAxis(brakePedal);
77 }
78
79 if (pedals.hasPedal(SimRacing::Clutch)) {
80 int clutchPedal = pedals.getPosition(SimRacing::Clutch, 0, ADC_Max);
81 Joystick.setZAxis(clutchPedal);
82 }
83
84 Joystick.sendState();
85}
